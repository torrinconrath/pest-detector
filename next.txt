# === IMPROVED ENVIRONMENTAL SIMULATION ===
class RealisticNightEnvironment:
    def apply_effect(self, img, mode):
        """Apply realistic nighttime effects"""
        # Convert to float for processing
        img = img.astype(np.float32)
        
        if mode == "clear_night":
            # Moderate night conditions
            img *= 0.4
            img += np.random.normal(0, 8, img.shape)
            
        elif mode == "foggy_night":
            # Heavy fog simulation
            img *= 0.3
            # Add fog overlay
            fog = np.ones_like(img) * 40
            alpha = random.uniform(0.3, 0.6)
            img = cv2.addWeighted(img, 1-alpha, fog, alpha, 0)
            img = cv2.GaussianBlur(img, (15, 15), 5)
            
        elif mode == "rainy_night":
            # Rain effects
            img *= 0.5
            # Add rain streaks
            for _ in range(50):
                x = random.randint(0, img.shape[1] - 3)
                length = random.randint(20, 60)
                cv2.line(img, (x, 0), (x, length), (30, 30, 30), 2)
            img = cv2.GaussianBlur(img, (7, 7), 2)
            
        elif mode == "lens_glare":
            # Lens glare and reflections
            img *= 0.6
            overlay = np.zeros_like(img)
            # Multiple glare sources
            for _ in range(random.randint(3, 6)):
                center = (random.randint(0, img.shape[1]), random.randint(0, img.shape[0]))
                radius = random.randint(40, 120)
                intensity = random.randint(100, 200)
                cv2.circle(overlay, center, radius, (intensity, intensity, intensity), -1)
                # Add bloom effect
                cv2.GaussianBlur(overlay, (51, 51), 30, overlay)
            img = cv2.addWeighted(img, 0.8, overlay, 0.4, 0)
            
        elif mode == "overcast":
            # Cloudy night
            img *= 0.7
            img = cv2.equalizeHist(img.astype(np.uint8)).astype(np.float32)
            
        elif mode == "pitch_black":
            # Very dark conditions
            img *= 0.2
            # Add some sensor noise
            img += np.random.normal(0, 12, img.shape)
            # Crush blacks
            img = np.power(img/255.0, 1.5) * 255

        # Always add some noise and clip
        img += np.random.normal(0, 6, img.shape)
        img = np.clip(img, 0, 255).astype(np.uint8)
        return img

# === IMPROVED RACCOON OVERLAY ===
class RealisticRaccoonOverlay:
    def __init__(self, frame_size):
        self.frame_size = frame_size
        self.images = self.load_and_augment_images()

    def load_and_augment_images(self):
        """Load raccoon images and apply realistic augmentations"""
        imgs = []
        for ext in ['*.jpg', '*.jpeg', '*.png']:
            imgs.extend(glob.glob(os.path.join(TEST_IMAGES_DIR, ext)))
        
        raccoons = []
        for img_path in imgs:
            img = cv2.imread(img_path)
            if img is not None:
                # Resize to reasonable dimensions first
                h, w = img.shape[:2]
                max_dim = max(h, w)
                if max_dim > 400:
                    scale = 400 / max_dim
                    img = cv2.resize(img, (int(w*scale), int(h*scale)))
                
                # Apply nighttime-like preprocessing
                img = self.preprocess_for_night(img)
                raccoons.append(img)
        
        print(f"✅ Loaded {len(raccoons)} raccoon images with night preprocessing.")
        return raccoons

    def preprocess_for_night(self, img):
        """Preprocess images to look more like nighttime captures"""
        # Reduce brightness and contrast
        img = img.astype(np.float32)
        img *= random.uniform(0.3, 0.7)  # Random brightness
        img += random.uniform(-10, 10)   # Random offset
        img = np.clip(img, 0, 255).astype(np.uint8)
        
        # Add slight blur to simulate night camera
        if random.random() > 0.5:
            img = cv2.GaussianBlur(img, (3, 3), 0.5)
            
        return img

    def overlay_raccoon(self, bg, attempt=0):
        """Realistic raccoon overlay with multiple attempts for good placement"""
        if not self.images or attempt > 5:
            return bg, None

        pest = random.choice(self.images).copy()
        h, w = pest.shape[:2]
        
        # More realistic size range for security cameras
        scale = random.uniform(0.15, 0.35)  # Smaller minimum size
        new_w, new_h = int(w * scale), int(h * scale)
        
        # Ensure minimum size
        new_w = max(new_w, 30)
        new_h = max(new_h, 30)
        
        pest_resized = cv2.resize(pest, (new_w, new_h))
        
        frame_h, frame_w = self.frame_size
        
        # Try to place in realistic locations (often lower part of frame)
        if random.random() > 0.3:  # 70% chance for ground-level placement
            y_min = int(frame_h * 0.6)  # Lower portion
            y_max = frame_h - new_h - 5
        else:
            y_min = 0
            y_max = frame_h - new_h - 5
            
        y_min = max(0, y_min)
        y_max = max(y_min, y_max)
        
        if y_max <= y_min:
            return self.overlay_raccoon(bg, attempt + 1)
            
        x = random.randint(10, max(11, frame_w - new_w - 10))
        y = random.randint(y_min, y_max)
        
        # Simple alpha blending for realism
        roi = bg[y:y+new_h, x:x+new_w]
        if roi.shape == pest_resized.shape:
            # Blend based on darkness (night-friendly)
            alpha = random.uniform(0.7, 0.9)
            blended = cv2.addWeighted(roi, 1-alpha, pest_resized, alpha, 0)
            bg[y:y+new_h, x:x+new_w] = blended
        
        return bg, (x, y, x + new_w, y + new_h)

# === IMPROVED METRICS & VISUALIZATION ===
def calculate_iou(box1, box2):
    """Calculate Intersection over Union"""
    if box1 is None or box2 is None:
        return 0.0
        
    x11, y11, x21, y21 = box1
    x12, y12, x22, y22 = box2
    
    xi1 = max(x11, x12)
    yi1 = max(y11, y12)
    xi2 = min(x21, x22)
    yi2 = min(y21, y22)
    
    inter_area = max(0, xi2 - xi1) * max(0, yi2 - yi1)
    box1_area = (x21 - x11) * (y21 - y11)
    box2_area = (x22 - x12) * (y22 - y12)
    union_area = box1_area + box2_area - inter_area
    
    return inter_area / (union_area + 1e-9)

def draw_detection_info(frame, detections, stats, frame_idx, mode, conf_thres, pest_bbox):
    """Improved visualization"""
    # Draw ground truth if exists
    if pest_bbox:
        x1, y1, x2, y2 = pest_bbox
        cv2.rectangle(frame, (x1, y1), (x2, y2), (255, 255, 0), 2)  # Cyan for ground truth
        cv2.putText(frame, "TRUTH", (x1, max(15, y1 - 6)),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.4, (255, 255, 0), 1)
    
    # Draw detections
    for (x1, y1, x2, y2, conf) in detections:
        color = (0, 255, 0)  # Green for correct detections
        # Check if this detection matches ground truth
        if pest_bbox:
            iou_val = calculate_iou((x1, y1, x2, y2), pest_bbox)
            if iou_val > 0.3:  # Good detection
                color = (0, 255, 0)  # Green
            else:  # False positive
                color = (0, 0, 255)  # Red
        
        cv2.rectangle(frame, (x1, y1), (x2, y2), color, 2)
        cv2.putText(frame, f"raccoon {conf:.2f}", (x1, max(15, y1 - 6)),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.4, color, 1)
    
    # Calculate current metrics
    current_recall = stats["detected"] / max(1, stats["true"])
    current_precision = stats["detected"] / max(1, stats["detected"] + stats["false_pos"])
    
    # Display info
    lines = [
        f"Frame: {frame_idx}/{stats['total']}",
        f"Environment: {mode}",
        f"Confidence: {conf_thres:.2f}",
        f"True Raccoons: {stats['true']}",
        f"Detected: {stats['detected']}",
        f"False Positives: {stats['false_pos']}",
        f"Current Recall: {current_recall:.3f}",
        f"Current Precision: {current_precision:.3f}"
    ]
    
    for i, text in enumerate(lines):
        y_pos = 20 + i * 16
        # Background for readability
        cv2.rectangle(frame, (5, y_pos-12), (250, y_pos+4), (0, 0, 0), -1)
        cv2.putText(frame, text, (10, y_pos),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.4, (255, 255, 255), 1)
    
    return frame
